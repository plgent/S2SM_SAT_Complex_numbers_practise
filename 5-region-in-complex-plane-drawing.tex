\begin{EnvFullwidth}
\begin{solutionorgrid}[.75in]
The shaded region is
\[
	\{z \in \C : \abs{z} \leq 1, \Im(z) \geq 0\}.
\]
\end{solutionorgrid}
\end{EnvFullwidth}

\part[3]
Show that if $\abs{z - 1} \geq \abs{z + 2}$, then $x \leq -\frac{1}{2}$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[2in]
On putting $z = x + iy$ there is
\begin{align*}
    \abs{x + iy - 1} &\geq \abs{x + iy + 2} \\
    (x - 1)^2 \cancel{+ y^2} &\geq (x + 2)^2 \cancel{+ y^2} \\
    \cancel{x^2} - 2x + 1 &\geq \cancel{x^2} + 4x + 4 \\
    -6x &\geq 3 \\
    x &\leq -\frac{1}{2}.
\end{align*}
\end{solutionorgrid}
\end{EnvFullwidth}
