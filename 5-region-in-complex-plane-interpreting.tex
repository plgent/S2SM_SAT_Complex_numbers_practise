\part[4]
If $(\conj{a - 3i})(1 - i) = b + i$, find the real numbers $a$ and $b$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[3in]
Let $z = (\conj{a - 3i})(1 - i)$ and let $w = b + i$. We have
\begin{align*}
	(\conj{a - 3i})(1 - i) &= (a + 3i)(1 - i) \\
	&= a - ai + 3i - 3i^2 \\
	&= a + 3 + i(3 - a) && (i^2 = -1).
\end{align*}
Now, $\Im(z) = 3 - a$ and $\Im(w) = 1$. Equating imaginary parts yields $3 - a = 1$. So, $a = 2$. Moreover, $\Re(z) = a + 3$ and $\Re(w) = b$. Thus, $b = 2 + 3$. So the solution is $a = 2$, $b = 5$.
\end{solutionorgrid}
\end{EnvFullwidth}

\part[4]
Write an inequality in $z$ that describes the shaded region in Figure~\ref{fig:region-in-the-complex-plane}.

\begin{EnvFullwidth}
\begin{solutionorgrid}[2in]
The circle has radius $1$ and is centred at the complex number $c = 1 - i$. Thus,
\[
    D = \{z \in \C : \abs{z - (1 - i)} \leq 1\}.
\]
\end{solutionorgrid}
\end{EnvFullwidth}
