\uplevel{
\begin{theorem}
\label{thm:sum-of-geometric-series}
Let $z \neq 1$ be a complex number and let $n$ be a positive integer. Then
\[
    1 + z + z^2 + \cdots + z^{n - 1} = \frac{z^n - 1}{z - 1}.
\]
\end{theorem}
}
