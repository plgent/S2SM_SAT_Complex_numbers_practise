Let $a$ be a real number and let $c$ be a complex number. The circle in Figure~\ref{fig:complex-number-geometry} is the set
\[
    S = \{z \in \C : \abs{z - c} = a\}.
\]
One endpoint of the diameter $OP$ of the circle is $P(2, 2)$.

\begin{figure}[h]
	\centering
	\begin{tikzpicture}

		yticklabels={},

		ylabel=$\Im$,

	\coordinate (O) at (axis cs:0,0);
	\coordinate (C) at (axis cs:1,1);
	\coordinate (P) at (axis cs:2,2);
	\addplot[] ({sqrt(2)*cos(x) + 1}, {sqrt(2)*sin(x) + 1}) node[midway, xshift=6pt] {$S$};
	\end{axis}
	\draw[semithick] (O) -- (P);
	\fill (O) circle (2pt) node[below left] {$O(0, 0)$} (P) circle (2pt) node[right] {$P(2, 2)$};

\begin{parts}

\part[2]
Find the value of $c$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[1.5in]
The centre $C$ of the circle is the midpoint of $OP$
\[
	C\left(\frac{0 + 2}{2}, \frac{0 + 2}{2}\right) = C(1, 1).
\]
Hence, $c = 1 + i$.
\end{solutionorgrid}
\end{EnvFullwidth}

\part[3]
Find the \emph{exact} value of $a$.

\begin{EnvFullwidth}
\begin{solutionorgrid}[3in]
Let $z = x + yi$. As $\abs{z - c} = a$ we have
\begin{align*}
	\abs{x + yi - (1 + i)} &= a \\
	\sqrt{(x - 1)^2 + (y - 1)^2} &= a \\
	(x - 1)^2 + (y - 1)^2 &= a^2.
\end{align*}
Thus, $a$ is the radius of the circle, centred at $(1, 1)$, so
\begin{align*}
	a &= \sqrt{(1 - 0)^2 + (1 - 0)^2} \\
	&= \sqrt{2}.
\end{align*}
\end{solutionorgrid}
\end{EnvFullwidth}

\end{parts}
